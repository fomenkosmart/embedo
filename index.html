embedo

A simple, fast, lightweight and standalone social media content embedder plugin for web

JS Implementation:

        
 /**
  * @example Embedo Implementation Sample
  *
  * @import https://cdn.rawgit.com/shobhitsharma/embedo/v0.2.2/embedo.min.js
  * @package npm install embedo --save
  * @package bower install embedo
  */
  
  import Embedo from './my/vendor/folder';
  
  const embedo = new Embedo({
    facebook: true, // Enable facebook SDK
    twitter: true,  // Enable twitter SDK
    instagram: true,  // Enable instagram SDK
    pinterest: true  // Enable pinterest SDK
  });

  // Calling .load() event [HTMLElement, url|string, options|object{}]
  embedo.load(
    document.getElementById('my-awesome-container'),
    'https://twitter.com/Sh0bhit/status/830544109857275906'
  );

  // Passing custom width and height as options
  embedo.load(
    document.getElementById('my-awesome-container'),
    'https://www.youtube.com/watch?v=5Ad3TH8XGSI',
    {
      width: 640,
      height: 480
    }
  );

  // Load without automagic (disables auto-resizing and scaling translations)
  embedo.load(
    document.getElementById('my-awesome-container'),
    'https://www.instagram.com/p/BQZV5JZllrH/',
    {
      strict: true
    }
  );

  // Refresh event can also be triggered
  document.addEventListener('resize', function (event) {
    embedo.refresh();
  });

  // DOM Watch Event
  document.getElementById('my-awesome-container').addEventListener('watch',
    function (event) {
      // Event log .detail contains updated element width and height
      console.log(event, event.detail);
    });
